// 水仙之爱绽于何方?
#include <bits/stdc++.h>

using f64 = double;
const f64 pi = acos(-1);
const f64 eps = 1e-8;

struct Point {
	f64 x, y;
	Point(f64 _x = 0, f64 _y = 0) : x(_x), y(_y)	{ }
};

using Vector = Point;
Vector operator + (Vector A, Vector B)	{return Vector(A.x + B.x, A.y + B.y);}
Vector operator - (Vector A, Vector B)	{return Vector(A.x - B.x, A.y - B.y);}
Vector operator * (Vector A, f64 p)	{return Vector(A.x * p, A.y * p);}
Vector operator / (Vector A, f64 p)	{return Vector(A.x / p, A.y / p);}

int dcmp(f64 x) {
	if (fabs(x) < eps)	return 0;
	else	return x < 0 ? -1 : 1;
}
int dcmp(f64 x, f64 y) {return dcmp(x - y);}

bool operator == (const Point& a, const Point& b) {
	return dcmp(a.x - b.x) == 0 && dcmp(a.y - b.y) == 0;
}

bool operator < (const Point& a, const Point& b) {
	return a.x == b.x ? a.y < b.y : a.x < b.x;
}

inline Vector trans(Point A, Point B) {return B - A;}
f64 Dot(Vector A, Vector B)	{return A.x * B.x + A.y * B.y;}
f64 Length(Vector A)	{return sqrt(Dot(A, A));}
f64 Angle(Vector A, Vector B)	{return acos(Dot(A, B) / Length(A) / Length(B));}
f64 PolarAngle(Vector A) {return atan2(A.y, A.x);}
f64 Cross(Vector A, Vector B) {return A.x * B.y - A.y * B.x;}
f64 Area2(Point A, Point B, Point C)	{return Cross(trans(A, B), trans(A, C));}

struct Line {
	Point p;
	Vector v;
	Line(Point _p, Vector _v) : p(_p), v(_v) { }
	Point point(f64 t) {return p + v * t;}
};

Vector Rotate(Vector A, f64 rad) {
	return Vector(A.x * cos(rad) - A.y * sin(rad), A.x * sin(rad) + A.y * cos(rad));
}

Vector Normal(Vector A) {
	f64 L = Length(A);
	return Vector(-A.y / L, A.x / L);
}

Point GetLineIntersection(Line l1, Line l2) {
	Point P = l1.p, Q = l2.p;
	Vector v = l1.v, w = l2.v;
	Vector u = trans(Q, P);
	f64 t = Cross(w, u) / Cross(v, w);
	return l1.point(t);
}

f64 DistanceToLine(Point P, Line l) {
	Vector v1 = l.v, v2 = trans(l.p, P);
	return fabs(Cross(v1, v2)) / Length(v1);
}

f64 DistanceToSegment(Point P, Point A, Point B) {
	if (A == B)	return Length(trans(A, P));
	Vector v1 = trans(A, B), v2 = trans(A, P), v3 = trans(B, P);
	if (Dot(v1, v2) < 0)	return Length(v2);
	if (Dot(v1, v3) > 0)	return Length(v3);
	return DistanceToLine(P, Line(A, v1));
}

Point GetLineProjection(Point P, Line l) {
	Vector v = l.v;
	Point A = l.p;
	f64 t0 = Dot(v, trans(A, P)) / Dot(v, v);
	return l.point(t0);
}

bool RapidRejection(Point A, Point B, Point C, Point D) {
	if (dcmp(std::min(A.x, B.x) - std::max(C.x, D.x)) > 0)	return false;
	if (dcmp(std::min(C.x, D.x) - std::max(A.x, B.x)) > 0)	return false;
	if (dcmp(std::min(A.y, B.y) - std::max(C.y, D.y)) > 0)	return false;
	if (dcmp(std::min(C.y, D.y) - std::max(A.y, B.y)) > 0)	return false;
	return true;
}

bool SegmentIntersection(Point A, Point B, Point C, Point D) {
	if (!RapidRejection(A, B, C, D))	return false;
	Vector v1 = trans(A, B), v2 = trans(C, D);
	return dcmp(Cross(v1, trans(A, C)) * Cross(v1, trans(A, D))) < 0 && dcmp(Cross(v2, trans(C, A)) * Cross(v2, trans(C, B))) < 0;
}

bool OnSegment(Point P, Point A, Point B) {
	return dcmp(Cross(trans(A, P), trans(B, P))) == 0 && dcmp(Dot(trans(A, P), trans(B, P))) < 0;
}

struct Circle {
	Point c;
	f64 r;
	Circle(Point _c, f64 _r) : c(_c), r(_r) { }
	Point point(f64 a) {
		return Point(c.x + cos(a) * r, c.y + sin(a) * r);
	}
};

int getLineCircleIntersection(Line l, Circle C, f64 &t1, f64 &t2, std::vector<Point>& sol) {
	f64 a = l.v.x, b = l.p.x - C.c.x, c = l.v.y, d = l.p.y - C.c.y;
	f64 e = a * a + c * c, f = 2 * (a * b + c * d), g = b * b + d * d - C.r * C.r;
	f64 delta = f * f - 4 * e * g;
	if (dcmp(delta) < 0)	return 0;
	if (dcmp(delta) == 0) {
		t1 = t2 = -f / (2 * e);
		sol.push_back(l.point(t1));
		return 1;
	}
	t1 = (-f - sqrt(delta)) / (2 * e);
	t2 = (-f + sqrt(delta)) / (2 * e);
	sol.push_back(l.point(t1));
	sol.push_back(l.point(t2));
	return 2;
}

int getCircleIntersection(Circle C1, Circle C2, std::vector<Point>& sol) {
	f64 d = Length(trans(C2.c, C1.c));
	if (dcmp(d) == 0) {
		if (dcmp(C1.r - C2.r) == 0)	return -1;
		return 0;
	}
	if (dcmp(C1.r + C2.r - d) < 0)	return 0;
	if (dcmp(fabs(C1.r - C2.r) - d) > 0)	return 0;

	f64 a = PolarAngle(trans(C1.c, C2.c));
	f64 da = acos((C1.r * C1.r + d * d - C2.r * C2.r) / (2 * C1.r * d));
	Point p1 = C1.point(a - da), p2 = C1.point(a + da);
	sol.push_back(p1);
	if (p1 == p2)	return 1;
	sol.push_back(p2);
	return 2;
}

// type = 0 切线, type = 1 找切点
// 都以 p 为原点
int getTangents(Point p, Circle C, std::vector<Vector> &v, int type) {
	Vector u = trans(p, C.c);
	f64 dist = Length(u);
	if (dcmp(dist, C.r) < 0)	return 0;
	else if (dcmp(dist, C.r) == 0)	{
		if (type == 0)
			v.push_back(Rotate(u, pi / 2));
		else
			v.push_back(Point(0, 0));
		return 1;
	} else {
		f64 A = fabs(asin(C.r / dist));
		if (type == 0) {
			v.push_back(Rotate(u, -A));
			v.push_back(Rotate(u, A));
		} else {
			Vector l1 = Rotate(u, -A), l2 = Rotate(u, A);
			v.push_back(l1 * cos(A));
			v.push_back(l2 * cos(A));
		}
		return 2;
	}
}

//a, b代表切线的两个切点坐标
int getCircleTangents(Circle A, Circle B, Point *a, Point *b) {
	int cnt = 0;
	if (A.r < B.r)	std::swap(A, B), std::swap(a, b);
	f64 d2 = (A.c.x - B.c.x) * (A.c.x - B.c.x) + (A.c.y - B.c.y) * (A.c.y - B.c.y);
	f64 rdiff = A.r - B.r;
	f64 rsum = A.r + B.r;
	//内含
	if (d2 < rdiff * rdiff)	return 0;
	//极角
	f64 base = PolarAngle(trans(A.c, B.c));
	if (d2 == 0 && A.r == B.r)	return -1;
	//内切
	if (d2 == rdiff * rdiff) {
		a[cnt] = A.point(base), b[cnt] = B.point(base), cnt++;
		return 1;
	}
	f64 ang = acos((A.r - B.r) / sqrt(d2));
	a[cnt] = A.point(base + ang), b[cnt] = B.point(base + ang), cnt++;
	a[cnt] = A.point(base - ang), b[cnt] = B.point(base - ang), cnt++;
	//外切
	if (d2 == rsum * rsum) {
		a[cnt] = A.point(base + ang), b[cnt] = B.point(pi + ang), cnt++;
	} else if(d2 > rsum * rsum) {
		f64 ang = acos((A.r + B.r) / sqrt(d2));
		a[cnt] = A.point(base + ang), b[cnt] = B.point(base + ang + pi), cnt++;
		a[cnt] = A.point(base - ang), b[cnt] = B.point(base - ang + pi), cnt++;
	}
	return cnt;
}

std::ifstream readin("beacons.in");
std::ofstream output("beacons.out");

std::vector<Point> hole;
std::vector<Circle> circ;
std::vector<int> fa;
inline int find(int x)	{return x == fa[x] ? x : fa[x] = find(fa[x]);}
inline void merge(int x, int y)	{
	x = find(x), y = find(y);
	if (x != y)
		fa[x] = y;
}

int n, m;

// 将负数极角变成正数
inline f64 reg(f64 x) {
	// 注意掉精度变成 eps 左右的情况……
	if (dcmp(x) <= 0)
		return x + 2 * pi;
	else
		return x;
}

struct sector {
	f64 st, ed;
	f64 length;
	// 大于
	bool operator < (const sector &b) const {
		return dcmp(length, b.length) == 1;
	}
};

void calc(int x) {
	// 所有可能用到的点, 平移到以 x 为原点
	std::vector<std::pair<Point, int>> ote;
	for (int i = 0; i < n; i++)
		if (hole[i] == hole[x])
			merge(i, x);
		else
			ote.push_back(std::make_pair(trans(hole[x], hole[i]), i));

	// 把所有点按极角排序
	std::sort(ote.begin(), ote.end(), [=] (const std::pair<Point, int> &a, const std::pair<Point, int> &b) {
			return dcmp(reg(PolarAngle(a.first)), 
						reg(PolarAngle(b.first)) ) == -1;
			});

	std::vector<sector> vec;

	for (int i = 0; i < m; i++) {
		std::vector<Point> T;
		getTangents(hole[x], circ[i], T, 1);
		if (dcmp(T[0].y) == -1 && dcmp(T[1].y) == 1) {
			// 如果横跨 x 轴，那么裁成两个扇形
			vec.push_back({reg(PolarAngle(T[0])), 2 * pi, Length(T[0])});
			vec.push_back({0, reg(PolarAngle(T[1])), Length(T[1])});
		} else {
			// 否则直接把它放进去
			f64 st = PolarAngle(T[0]);
			f64 ed = PolarAngle(T[1]);
			// st 为 0 的时候不能 +2pi, 但是 ed 要
			if (dcmp(ed) <=  0)	ed += 2 * pi;
			if (dcmp(st) == -1)	st += 2 * pi;
			st -= eps;
			ed += eps;
			vec.push_back({st, ed, Length(T[0])});
		}
	}

	// 按起始线的极角排序
	std::sort(vec.begin(), vec.end(), [=] (const sector &a, const sector &b) {
			return dcmp(a.st, b.st) == -1;
			});

	std::priority_queue<sector> q;
	size_t j = 0;
	for (const auto &t : ote) {
		auto p = t.first;
		auto num = t.second;

		auto Angle = reg(PolarAngle(p));
		while (j < vec.size() && dcmp(vec[j].st, Angle) <= 0)
			q.push(vec[j++]);

		while (!q.empty() && dcmp(q.top().ed, Angle) == -1)
			q.pop();

		f64 len = 1e18;
		if (!q.empty())
			len = q.top().length;
		if (dcmp(Length(p), len) == -1)
			merge(num, x);
	}
}

int main() {
	readin >> n >> m;
	hole.reserve(n), circ.reserve(m);
	fa.resize(n);


	for (int i = 0; i < n; i++) {
		int x, y;
		readin >> x >> y;
		hole.emplace_back(x, y);
		fa[i] = i;
	}
	for (int i = 0; i < m; i++) {
		int x, y, r;
		readin >> x >> y >> r;
		circ.emplace_back(Point(x, y), r);
	}
	for (int i = 0; i < n; i++)
		calc(i);
	int ans = 0;
	for (int i = 0; i < n; i++)
		if (fa[i] == i)
			ans++;
	output << ans - 1 << '\n';
	return 0;
}
